\begin{defnbox}\nospacing
  \begin{defn}[Greedy Policy]\label{defn:mdp_greedy_policy}\leavevmode\\
    \imp{Assuming} we know $\valfu^{\policy_{t-1}}$ then we could choose a greedy policy:
    \begin{align}
      \action^{*}&=\policy_{t}(x)\\[-1\jot]
      :&=\argmax_{\action\in\Asp}\reward(x,\action)+\gammac\sum_{x'\in\Ssp}\Prob\left(x'|x,\action\right)\valfu^{\policy_{t-1}}\left(x'\right)\nonumber
    \end{align}
  \end{defn}
\end{defnbox}
